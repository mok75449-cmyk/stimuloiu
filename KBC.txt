        function addReviewToFeed(name, text) {
            const review = createReviewElement(name, text);
            feedArea.appendChild(review);
            feedArea.scrollTop = feedArea.scrollHeight;
            if (feedArea.children.length > 30) {
                feedArea.removeChild(feedArea.firstChild);
            }
        }

        function scheduleNextReview() {
            const delay = Math.floor(Math.random() * 3000) + 1500;
            setTimeout(() => {
                const text = reviewDatabase[reviewIndex % reviewDatabase.length];
                const name = names[Math.floor(Math.random() * names.length)];
                addReviewToFeed(name, text);
                reviewIndex++;
                scheduleNextReview();
            }, delay);
        }
